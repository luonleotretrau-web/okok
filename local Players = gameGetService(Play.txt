function Notify(title, message, duration)
	duration = duration or 3

	local notif = Instance.new("Frame")
	notif.Size = UDim2.new(1, 0, 0, 70)
	notif.Position = UDim2.new(0, 360, 0, 0) -- fora da tela à direita
	notif.BackgroundColor3 = Color3.fromRGB(20, 20, 25)
	notif.BorderSizePixel = 0
	notif.ClipsDescendants = true
	notif.Parent = notificationHolder

	local notifCorner = Instance.new("UICorner")
	notifCorner.CornerRadius = UDim.new(0, 8)
	notifCorner.Parent = notif
	local notifStroke = Instance.new("UIStroke")
	notifStroke.Color = Color3.fromRGB(40, 40, 40)
	notifStroke.Thickness = 1
	notifStroke.Parent = notif

	local titleLabel = Instance.new("TextLabel")
	titleLabel.Size = UDim2.new(1, -30, 0, 20)
	titleLabel.Position = UDim2.new(0, 12, 0, 6)
	titleLabel.BackgroundTransparency = 1
	titleLabel.Text = title
	titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
	titleLabel.Font = Enum.Font.GothamBold
	titleLabel.TextSize = 13
	titleLabel.TextXAlignment = Enum.TextXAlignment.Left
	titleLabel.Parent = notif

	local closeNotif = Instance.new("TextButton")
	closeNotif.Size = UDim2.new(0, 22, 0, 22)
	closeNotif.Position = UDim2.new(1, -28, 0, 4)
	closeNotif.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
	closeNotif.BorderSizePixel = 0
	closeNotif.Text = "X"
	closeNotif.TextColor3 = Color3.fromRGB(255, 255, 255)
	closeNotif.Font = Enum.Font.GothamBold
	closeNotif.TextSize = 12
	closeNotif.Parent = notif
	local closeCorner = Instance.new("UICorner")
	closeCorner.CornerRadius = UDim.new(0, 4)
	closeCorner.Parent = closeNotif
	closeNotif.MouseButton1Click:Connect(function()
		notif:Destroy()
		rearrange()
	end)

	local msgLabel = Instance.new("TextLabel")
	msgLabel.Size = UDim2.new(1, -24, 0, 30)
	msgLabel.Position = UDim2.new(0, 12, 0, 28)
	msgLabel.BackgroundTransparency = 1
	msgLabel.Text = message
	msgLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
	msgLabel.Font = Enum.Font.Gotham
	msgLabel.TextSize = 12
	msgLabel.TextXAlignment = Enum.TextXAlignment.Left
	msgLabel.TextWrapped = true
	msgLabel.Parent = notif

	-- Animação de entrada
	local slideIn = TweenService:Create(notif, TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = UDim2.new(0, 0, 0, 0)})
	slideIn:Play()

	local function rearrange()
		local yOffset = 0
		local children = notificationHolder:GetChildren()
		local notifFrames = {}
		for _, child in ipairs(children) do
			if child:IsA("Frame") then
				table.insert(notifFrames, child)
			end
		end
		for i, nf in ipairs(notifFrames) do
			local targetPos = UDim2.new(0, 0, 0, yOffset)
			TweenService:Create(nf, TweenInfo.new(0.3, Enum.EasingStyle.Quad), {Position = targetPos}):Play()
			yOffset = yOffset + nf.Size.Y.Offset + 6
		end
	end

	rearrange()

	delay(duration, function()
		if notif and notif.Parent then
			-- Animação de saída
			local slideOut = TweenService:Create(notif, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Position = UDim2.new(0, 360, 0, notif.Position.Y.Offset)})
			slideOut:Play()
			slideOut.Completed:Connect(function()
				notif:Destroy()
				rearrange()
			end)
		end
	end)
end

	local title = config.Title or config.Text or "Keybind"
	local default = config.Default or "None"

	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(1, -16, 0, 42)
	frame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)

	stroke.Thickness = 1

	local label = Instance.new("TextLabel", frame)
	label.Size = UDim2.new(0, 110, 1, 0)
	label.Position = UDim2.new(0, 12, 0, 0)
	label.BackgroundTransparency = 1
	label.Text = title
	label.TextColor3 = Color3.fromRGB(255, 255, 255)
	label.Font = Enum.Font.GothamMedium
	label.TextSize = 12
	label.TextXAlignment = Enum.TextXAlignment.Left

	local btn = Instance.new("TextButton", frame)
	btn.Size = UDim2.new(0, 80, 0, 28)
	btn.Position = UDim2.new(1, -92, 0.5, -14)
	btn.BackgroundColor3 = Color3.fromRGB(28, 28, 33)
	btn.BorderSizePixel = 0
	btn.Text = default
	btn.TextColor3 = Color3.fromRGB(255, 255, 255)
	btn.Font = Enum.Font.GothamBold
	btn.TextSize = 12
	local btnCorner = Instance.new("UICorner", btn)
	btnCorner.CornerRadius = UDim.new(0, 6)
	local btnStroke = Instance.new("UIStroke", btn)
	btnStroke.Color = Color3.fromRGB(50, 50, 55)
	btnStroke.Thickness = 1

	local currentKey = default
	local waiting = false

	local function setKey(key)
		currentKey = key
		btn.Text = key
		if config.Callback then
			config.Callback(key)
		end
		waiting = false
	end

	btn.MouseButton1Click:Connect(function()
		if waiting then return end
		waiting = true
		btn.Text = "..."
		local connection
		connection = game:GetService("UserInputService").InputBegan:Connect(function(input, gameProcessed)
			if gameProcessed then return end
			if waiting then
				if input.KeyCode ~= Enum.KeyCode.Unknown then
					local keyName = input.KeyCode.Name
					if keyName == "Backspace" then
						setKey("None")
					else
						setKey(keyName)
					end
				elseif input.UserInputType == Enum.UserInputType.MouseButton1 then
					setKey("Mouse1")
				elseif input.UserInputType == Enum.UserInputType.MouseButton2 then
					setKey("Mouse2")
				elseif input.UserInputType == Enum.UserInputType.Touch then
					setKey("Touch")
				end
				connection:Disconnect()
			end
		end)
		task.delay(5, function()
			if waiting then
				waiting = false
				btn.Text = currentKey
				connection:Disconnect()
			end
		end)
	end)

	return frame
end

	listScroller.ScrollBarThickness = 4
	listScroller.ScrollBarImageColor3 = Color3.fromRGB(120, 120, 120)
	listScroller.AutomaticCanvasSize = Enum.AutomaticSize.Y
	listScroller.CanvasSize = UDim2.new(0, 0, 0, 0)
	listScroller.ZIndex = 1000

	local listLayout = Instance.new("UIListLayout", listScroller)
	listLayout.Padding = UDim.new(0, 2)
	listLayout.SortOrder = Enum.SortOrder.LayoutOrder

	local listPadding = Instance.new("UIPadding", listScroller)
	listPadding.PaddingTop = UDim.new(0, 3)
	listPadding.PaddingBottom = UDim.new(0, 3)
	listPadding.PaddingLeft = UDim.new(0, 3)
	listPadding.PaddingRight = UDim.new(0, 3)

	local currentOptions = {}
	if options then
		for i, v in ipairs(options) do
			table.insert(currentOptions, v)
		end
	end
	local isOpen = false

	-- Posiciona a lista logo abaixo do botão
	local function updateListPosition()
		local absPos = mainBtn.AbsolutePosition
		local absSize = mainBtn.AbsoluteSize
		listContainer.Position = UDim2.new(0, absPos.X, 0, absPos.Y + absSize.Y + 4)
	end

	-- Preenche a lista com os itens
	local function rebuildList()
		-- Remove apenas botões antigos
		for _, child in ipairs(listScroller:GetChildren()) do
			if child:IsA("TextButton") then
				child:Destroy()
			end
		end

		-- Cria os novos itens
		for _, option in ipairs(currentOptions) do
			local item = Instance.new("TextButton")
			item.Size = UDim2.new(1, -6, 0, 28)
			item.BackgroundColor3 = Color3.fromRGB(35, 35, 40)
			item.BorderSizePixel = 0
			item.Text = option
			item.TextColor3 = Color3.fromRGB(255, 255, 255)
			item.Font = Enum.Font.Gotham
			item.TextSize = 12
			item.Parent = listScroller
			Instance.new("UICorner", item).CornerRadius = UDim.new(0, 4)
			item.ZIndex = 1000    -- garante que os itens fiquem sobre tudo

			item.MouseButton1Click:Connect(function()
				mainBtn.Text = option
				listContainer.Visible = false
				isOpen = false
				if callback then callback(option) end
			end)
		end

		-- Truque para forçar o recálculo do canvas (Automático)
		listScroller.AutomaticCanvasSize = Enum.AutomaticSize.None
		listScroller.AutomaticCanvasSize = Enum.AutomaticSize.Y
		task.wait(0.01)  -- pequena pausa para o motor processar
		listScroller.CanvasSize = UDim2.new(0, 0, 0, 0)
	end

	-- Abre / fecha
	mainBtn.MouseButton1Click:Connect(function()
		if isOpen then
			listContainer.Visible = false
			isOpen = false
		else
			isOpen = true
			rebuildList()
			updateListPosition()
			listContainer.Visible = true
			-- Trazer o container para a frente (re-parente para ser o último filho)
			listContainer.Parent = gui
		end
	end)

	-- Fecha ao clicar fora
	local function onGlobalClick(input)
		if not isOpen then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			local mousePos = UserInputService:GetMouseLocation()
			local inList = (
				mousePos.X >= listContainer.AbsolutePosition.X and
				mousePos.X <= listContainer.AbsolutePosition.X + listContainer.AbsoluteSize.X and
				mousePos.Y >= listContainer.AbsolutePosition.Y and
				mousePos.Y <= listContainer.AbsolutePosition.Y + listContainer.AbsoluteSize.Y
			)
			local inBtn = (
				mousePos.X >= mainBtn.AbsolutePosition.X and
				mousePos.X <= mainBtn.AbsolutePosition.X + mainBtn.AbsoluteSize.X and
				mousePos.Y >= mainBtn.AbsolutePosition.Y and
				mousePos.Y <= mainBtn.AbsolutePosition.Y + mainBtn.AbsoluteSize.Y
			)
			if not inList and not inBtn then
				listContainer.Visible = false
				isOpen = false
			end
		end
	end
	UserInputService.InputBegan:Connect(onGlobalClick)

	return frame
end

	function tab:CreateSlider(text, min, max, default, callback)
		local sliderFrame = Instance.new("Frame")
		sliderFrame.Size = UDim2.new(1, -16, 0, 56)
		sliderFrame.BackgroundTransparency = 1
		sliderFrame.Parent = tabContent

		local label = Instance.new("TextLabel")
		label.Size = UDim2.new(1, 0, 0, 16)
		label.BackgroundTransparency = 1
		label.Text = text .. ": " .. tostring(default)
		label.TextColor3 = Color3.fromRGB(220, 220, 220)
		label.Font = Enum.Font.GothamMedium
		label.TextSize = 12
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.Parent = sliderFrame

		-- Barra de fundo com clique
		local sliderBar = Instance.new("TextButton")
		sliderBar.Size = UDim2.new(1, -20, 0, 8)
		sliderBar.Position = UDim2.new(0, 10, 0, 30)
		sliderBar.BackgroundColor3 = Color3.fromRGB(45, 45, 50)
		sliderBar.BorderSizePixel = 0
		sliderBar.Text = ""
		sliderBar.AutoButtonColor = false
		sliderBar.Parent = sliderFrame
		local barCorner = Instance.new("UICorner")
		barCorner.CornerRadius = UDim.new(1, 0)
		barCorner.Parent = sliderBar
		local barStroke = Instance.new("UIStroke")
		barStroke.Color = Color3.fromRGB(60, 60, 65)
		barStroke.Thickness = 1
		barStroke.Parent = sliderBar

		-- Preenchimento
		local fill = Instance.new("Frame")
		fill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
		fill.BackgroundColor3 = Color3.fromRGB(255, 70, 70)
		fill.BorderSizePixel = 0
		fill.Parent = sliderBar
		local fillCorner = Instance.new("UICorner")
		fillCorner.CornerRadius = UDim.new(1, 0)
		fillCorner.Parent = fill

		-- Knob
		local knob = Instance.new("TextButton")
		knob.Size = UDim2.new(0, 18, 0, 18)
		knob.Position = UDim2.new((default - min) / (max - min), -9, 0.5, -9)
		knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
		knob.BorderSizePixel = 0
		knob.Text = ""
		knob.AutoButtonColor = false
		knob.Parent = sliderBar
		local knobCorner = Instance.new("UICorner")
		knobCorner.CornerRadius = UDim.new(1, 0)
		knobCorner.Parent = knob
		local knobStroke = Instance.new("UIStroke")
		knobStroke.Color = Color3.fromRGB(180, 180, 180)
		knobStroke.Thickness = 1
		knobStroke.Parent = knob

		local value = default
		local dragging = false

		local function getValueFromPosition(mouseX)
			local barPos = sliderBar.AbsolutePosition.X
			local barWidth = sliderBar.AbsoluteSize.X
			if barWidth > 0 then
				local percent = math.clamp((mouseX - barPos) / barWidth, 0, 1)
				return math.floor(min + (max - min) * percent)
			end
			return value
		end

		local function updateSlider(newValue, instant)
			value = math.clamp(newValue, min, max)
			local relativeX = (value - min) / (max - min)
			if instant then
				-- sem animação (arrastando)
				fill.Size = UDim2.new(relativeX, 0, 1, 0)
				knob.Position = UDim2.new(relativeX, -9, 0.5, -9)
			else
				-- animação suave
				TweenService:Create(fill, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(relativeX, 0, 1, 0)}):Play()
				TweenService:Create(knob, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = UDim2.new(relativeX, -9, 0.5, -9)}):Play()
			end
			label.Text = text .. ": " .. value
			if callback then
				callback(value)
			end
		end

		-- Eventos de interação
		local function onInputBegin(input)
			if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
				dragging = true
				local newValue = getValueFromPosition(input.Position.X)
				updateSlider(newValue, true) -- instantâneo enquanto arrasta
			end
		end

		local function onInputMove(input)
			if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
				local newValue = getValueFromPosition(input.Position.X)
				updateSlider(newValue, true) -- instantâneo enquanto arrasta
			end
		end

		local function onInputEnd(input)
			if dragging and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
				dragging = false
				-- anima para a posição final (opcional, já está na posição)
				updateSlider(value, false) -- com animação
			end
		end

		sliderBar.InputBegan:Connect(onInputBegin)
		fill.InputBegan:Connect(onInputBegin)
		knob.InputBegan:Connect(onInputBegin)
		UserInputService.InputChanged:Connect(onInputMove)
		UserInputService.InputEnded:Connect(onInputEnd)

		updateSlider(default, false)
		return sliderFrame
	end

	-- Lógica de seleção da aba
	tabButton.MouseButton1Click:Connect(function()
		for _, t in pairs(tabs) do
			t.Content.Visible = false
			-- resetar estilo de todos os botões
			TweenService:Create(t.Button, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(28, 28, 33), TextColor3 = Color3.fromRGB(200, 200, 200)}):Play()
		end
		tabContent.Visible = true
		-- destaque no botão ativo
		TweenService:Create(tabButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(50, 20, 20), TextColor3 = Color3.fromRGB(255, 255, 255)}):Play()
	end)

	table.insert(tabs, tab)
	return tab
end

-- Função da barra de vida (quadrada acima da cabeça)
local function updateHealthBar(plr, character)
    local humanoid = character:FindFirstChild("Humanoid")
    local head = character:FindFirstChild("Head")
    if not (humanoid and head) then return end

    local oldBar = head:FindFirstChild("HealthBar")
    if oldBar then oldBar:Destroy() end

    local billboard = Instance.new("BillboardGui")
    billboard.Name = "HealthBar"
    billboard.AlwaysOnTop = true
    billboard.Size = UDim2.new(0, 80, 0, 8)
    billboard.StudsOffset = Vector3.new(0, 2.2, 0)
    billboard.Parent = head

    local background = Instance.new("Frame")
    background.Size = UDim2.new(1, 0, 1, 0)
    background.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
    background.BorderSizePixel = 0
    background.Parent = billboard
    Instance.new("UICorner", background).CornerRadius = UDim.new(0, 2)

    local fill = Instance.new("Frame")
    fill.BorderSizePixel = 0
    fill.Parent = background
    Instance.new("UICorner", fill).CornerRadius = UDim.new(0, 2)

    local function updateFill()
        local percent = math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1)
        fill.Size = UDim2.new(percent, 0, 1, 0)
        if percent <= 0.2 then
            fill.BackgroundColor3 = Color3.fromRGB(255, 0, 0)   -- vermelho
        elseif percent <= 0.5 then
            fill.BackgroundColor3 = Color3.fromRGB(255, 255, 0) -- amarelo
        else
            fill.BackgroundColor3 = Color3.fromRGB(0, 255, 0)   -- verde
        end
    end
    updateFill()
    humanoid.HealthChanged:Connect(updateFill)
end

-- Cria ESP para um jogador (persistente e auto-respawn)
local function createESPForPlayer(plr)
    task.spawn(function()
        while plr.Parent and espEnabled do
            local char = plr.Character
            if char and plr ~= lp then
                -- Highlight (cor configurável)
                local hl = char:FindFirstChild("c00l_ESP")
                if not hl then
                    hl = Instance.new("Highlight", char)
                    hl.Name = "c00l_ESP"
                    hl.FillTransparency = 0.5
                    hl.OutlineTransparency = 0.2
                end
                hl.FillColor = espColor

                -- Name tag (se habilitado)
                if nameTagsEnabled then
                    local head = char:FindFirstChild("Head")
                    if head and not head:FindFirstChild("c00l_Name") then
                        local bill = Instance.new("BillboardGui", head)
                        bill.Name = "c00l_Name"
                        bill.AlwaysOnTop = true
                        bill.Size = UDim2.new(0, 100, 0, 50)
                        bill.StudsOffset = Vector3.new(0, 1.5, 0)
                        local label = Instance.new("TextLabel", bill)
                        label.Size = UDim2.new(1, 0, 1, 0)
                        label.BackgroundTransparency = 1
                        label.Text = plr.Name
                        label.TextColor3 = Color3.new(1, 1, 1)
                        label.TextScaled = true
                        label.Parent = bill
                    end
                elseif char:FindFirstChild("Head") then
                    local tag = char.Head:FindFirstChild("c00l_Name")
                    if tag then tag:Destroy() end
                end

                -- Barra de vida (se habilitado)
                if healthBarEnabled then
                    local head = char:FindFirstChild("Head")
                    if head and not head:FindFirstChild("HealthBar") then
                        updateHealthBar(plr, char)
                    end
                end
            end
            task.wait(0.3)
        end
    end)
end

-- Liga/desliga tudo (highlight + name tags + barra)
function setESP(enabled)
    espEnabled = enabled
    if enabled then
        for _, p in pairs(Players:GetPlayers()) do
            createESPForPlayer(p)
        end
        if not _espConnection then
            _espConnection = Players.PlayerAdded:Connect(function(p)
                if espEnabled then createESPForPlayer(p) end
            end)
        end
    else
        for _, p in pairs(Players:GetPlayers()) do
            if p.Character then
                local hl = p.Character:FindFirstChild("c00l_ESP")
                if hl then hl:Destroy() end
                if p.Character:FindFirstChild("Head") then
                    local tag = p.Character.Head:FindFirstChild("c00l_Name")
                    if tag then tag:Destroy() end
                    local bar = p.Character.Head:FindFirstChild("HealthBar")
                    if bar then bar:Destroy() end
                end
            end
        end
    end
end

local function teleportToPlayer(name)
    if name == nil or name == "" then
        Notify("Teleport", "No player selected", 2)
        return
    end
    local target = Players:FindFirstChild(name)
    if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
        local myChar = lp.Character
        if myChar and myChar:FindFirstChild("HumanoidRootPart") then
            myChar.HumanoidRootPart.CFrame = target.Character.HumanoidRootPart.CFrame * CFrame.new(0, 3, 0)
            Notify("Teleport", "Teleported to " .. name, 2)
        else
            Notify("Teleport", "Your character is not ready", 2)
        end
    else
        Notify("Teleport", "Invalid target or no character", 2)
    end
end

-- Activate all spells once
local function equipAllSpells()
    local allSpells = {}
    for _, spellList in pairs(powers) do
        for _, spell in ipairs(spellList) do
            table.insert(allSpells, spell)
        end
    end
    for _, skill in ipairs(rebirthSkills) do
        table.insert(allSpells, skill)
    end

    local index = 1
    local function equipNext()
        if index > #allSpells then
            Notify("Spells", "All spells activated! (" .. #allSpells .. " items)", 3)
            return
        end
        local spell = allSpells[index]
        if Remote then
            Remote:FireServer("equip_mystery_spell", spell)
        end
        index = index + 1
        task.wait(0.05)
        equipNext()
    end
    equipNext()
end

-- Speed / Jump functions
local function setWalkSpeed(speed)
    currentSpeed = math.clamp(speed, 16, 500)
    if lockSpeed and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.WalkSpeed = currentSpeed
    end
end

local function setJumpPower(power)
    currentJump = math.clamp(power, 50, 500)
    if lockJump and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.JumpPower = currentJump
    end
end

local function lockWalkSpeed(lock)
    lockSpeed = lock
    if lock and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.WalkSpeed = currentSpeed
    elseif not lock and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.WalkSpeed = 16
    end
end

local function lockJumpPower(lock)
    lockJump = lock
    if lock and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.JumpPower = currentJump
    elseif not lock and lp.Character and lp.Character:FindFirstChild("Humanoid") then
        lp.Character.Humanoid.JumpPower = 50
    end
end
